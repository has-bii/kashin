1. Build the N-month window. Create a helper function `buildMonthBuckets(months: number)` at module scope (not exported):
   ```typescript
   function buildMonthBuckets(months: number): string[] {
     const result: string[] = []
     const now = new Date()
     for (let i = months - 1; i >= 0; i--) {
       const d = new Date(now.getFullYear(), now.getMonth() - i, 1)
       result.push(`${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}`)
     }
     return result
   }
   ```

2. Compute the date range for the full window:
   ```typescript
   const now = new Date()
   const windowStart = new Date(now.getFullYear(), now.getMonth() - (months - 1), 1)
   const windowEnd = new Date(now.getFullYear(), now.getMonth() + 1, 0)
   ```

3. Fetch all transactions in the window:
   ```typescript
   const transactions = await prisma.transaction.findMany({
     where: {
       userId,
       transactionDate: { gte: windowStart, lte: windowEnd },
     },
     select: { type: true, amount: true, transactionDate: true },
   })
   ```

4. Reduce into a Map keyed by "YYYY-MM":
   ```typescript
   const bucketMap = new Map<string, { income: number; expense: number }>()
   for (const tx of transactions) {
     const key = `${tx.transactionDate.getFullYear()}-${String(tx.transactionDate.getMonth() + 1).padStart(2, "0")}`
     const entry = bucketMap.get(key) ?? { income: 0, expense: 0 }
     const amount = parseFloat(tx.amount.toString())
     if (tx.type === "income") entry.income += amount
     else entry.expense += amount
     bucketMap.set(key, entry)
   }
   ```

5. Build the final response array using `buildMonthBuckets(months)`, zero-filling missing months:
   ```typescript
   const monthKeys = buildMonthBuckets(months)
   return monthKeys.map((month) => ({
     month,
     income: bucketMap.get(month)?.income ?? 0,
     expense: bucketMap.get(month)?.expense ?? 0,
   }))
   ```
